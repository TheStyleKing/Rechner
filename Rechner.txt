package rechner;

import java.awt.event.KeyEvent;
import java.util.Scanner;

public class Rechner {

    /**
     * @param args the command line arguments
     */
    
    static int art;
    static int key;
    
    static boolean korrekt = false;
    static boolean rechnerAktiv = true;
    public static void main(String[] args) {
        // TODO code application logic here
        
        Scanner eingabe = new Scanner(System.in);
        
        System.out.println("Rechner");
        while(rechnerAktiv)
        {
            if(!korrekt)
            {
                System.out.println("Wie lautet deine erste Zahl?");
                double zahl1 = eingabe.nextDouble();
                System.out.println("Wie lautet deine zweite Zahl?");
                double zahl2 = eingabe.nextDouble();
                System.out.println("Wie willst du rechnen? (ADD = 0 / SUBB = 1)");
                art = eingabe.nextInt();

                double erg;
                erg = rechnen(zahl1,zahl2);
                System.out.println("__________________");
                if(erg < 0)
                {
                    System.out.println("HINWEIS: Du bist im MINUS Bereich.");
                }
            }
            
                       
            System.out.println("MÃ¶chtest du noch einmal rechnen?");
            int wahl = eingabe.nextInt();
            if(wahl == 0)
            {
                rechnerAktiv = false;
            }
            else if(wahl == 1)
            {
                rechnerAktiv = true;
                korrekt = false;
            }
        }
    }
    static double rechnen(double zahl1,double zahl2)
    {
        double erg = 0;
        if(art == 0)
        {
            erg = zahl1 + zahl2;  
            korrekt = true;
            System.out.println("Ergebnis:"+erg);
        }
        else if(art == 1)
        {
            erg = zahl1 - zahl2;
            korrekt = true;
            System.out.println("Ergebnis:"+erg);
        }         
        else if(art > 1)
        {
            System.out.println("Die Zahl " +art+" hat keine Zuweisung!");
            System.out.println("Versuche es erneut.");
            erg = 0;
        }
        return erg;
    }
    
    
}
